	\begin{definition}[导数]
		函数 $f(x)$ 在点 $x_0$ 处的导数定义为：
		\[
		f'(x_0) = \lim_{h \to 0} \frac{f(x_0 + h) - f(x_0)}{h}
		\]
	\end{definition}
	
	\begin{theorem}[勾股定理]
		在直角三角形中，斜边的平方等于两直角边的平方和：
		\[
		a^2 + b^2 = c^2
		\]
	\end{theorem}
	
	\begin{proof}
		通过面积法可以证明此定理。
	\end{proof}
